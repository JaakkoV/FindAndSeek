1   package dev.jaakkovirtanen.findandseek.ui;
2   
3   import dev.jaakkovirtanen.findandseek.levels.*;
4   import dev.jaakkovirtanen.findandseek.mapobjects.*;
5   import java.awt.*;
6   import javax.swing.*;
7   
8   public class SidePanel extends JPanel {
9   
10      private final Player p;
11      private final Level l;
12      private JLabel goalsHit;
13      private JLabel optimal;
14      private JLabel optimalCumulative;
15      private JLabel playerMoves;
16      private JLabel playerMovesSinceHit;
17      private JLabel movesBehindTheOptimal;
18      private JLabel percentageOfExtraMoves;
19  
20      /**
21       * Constructor for SidePanel, needs gui.
22       *
23       * @param gui Gui, which wants SidePanel to be drawn
24       */
25      public SidePanel(GUI gui) {
26          this.setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
27          this.p = gui.getGame().getGameBoard().getPlayer();
28          this.l = gui.getGame().getGameBoard().getLevel();
29          createComponents();
30      }
31  
32      @Override
33      protected void paintComponent(Graphics grphcs) {
34          super.paintComponent(grphcs);
35          this.setBackground(Color.WHITE);
36          setGoalsHit(l.getHowManyGoals());
37          setOptimalMoves(l.getOptimalMoves());
38          setCumulativeOptimalMoves(l.getOptimalMovesCumulative());
39          setPlayerMoves();
40          setPlayerMovesSinceHit();
41          setBehindOptimal();
42          setPercentageOfExtraMoves();
43          repaint();
44      }
45  
46      private void createComponents() {
47          String stars = "***********************************";
48          JLabel empty = new JLabel(stars);
49  
50          JLabel gameStats = new JLabel("Game Statistics");
51          gameStats.setFont(new Font("Courier New", Font.BOLD, 18));
52          add(gameStats);
53          add(empty);
54  
55          this.goalsHit = new JLabel("Goals: 0");
56          add(goalsHit);
57  
58          this.playerMoves = new JLabel("Total moves: 0");
59          add(playerMoves);
60  
61          this.optimalCumulative = new JLabel("Cumulative optimal 0");
62          add(optimalCumulative);
63  
64          this.playerMovesSinceHit = new JLabel("Moves from last goal: 0");
65          add(playerMovesSinceHit);
66  
67          this.optimal = new JLabel("optimal: " + l.getOptimalMoves());
68          add(optimal);
69  
70          this.movesBehindTheOptimal = new JLabel("You are 0 behind the optimal");
71          add(movesBehindTheOptimal);
72          JLabel secEmpty = new JLabel(stars);
73          add(secEmpty);
74  
75          JLabel percentageBelowThis = new JLabel("Additional moves %: ");
76          percentageBelowThis.setFont(new Font("Courier New", Font.BOLD, 18));
77          add(percentageBelowThis);
78  
79          this.percentageOfExtraMoves = new JLabel();
80          this.percentageOfExtraMoves.setFont(new Font("Courier New", Font.BOLD, 20));
81          this.percentageOfExtraMoves.setPreferredSize(new Dimension(100, 100));
82          this.percentageOfExtraMoves.setForeground(Color.red);
83          add(percentageOfExtraMoves);
84          JLabel thirdEmpty = new JLabel(stars);
85          add(thirdEmpty);
86  
87      }
88  
89      /**
90       * Asks SidePanel to update GoalsHit-gameStats.
91       *
92       * GUI uses this
93       *
94       * @param goalsHit Which number to be set
95       */
96      public void setGoalsHit(int goalsHit) {
97          this.goalsHit.setText("Goals hit: " + goalsHit);
98      }
99  
100     /**
101      * Asks SidePanel to update OptimalMoves-gameStats.
102      *
103      * GUI uses this
104      *
105      * @param optimal Which number to be set
106      */
107     public void setOptimalMoves(int optimal) {
108         this.optimal.setText("Optimal distance to next: " + optimal);
109     }
110 
111     /**
112      * Asks SidePanel to update CumulativeOptimal-gameStats.
113      *
114      * GUI uses this
115      *
116      * @param cumuOptimal Which number to be set
117      */
118     public void setCumulativeOptimalMoves(int cumuOptimal) {
119         this.optimalCumulative.setText("Cumulative optimal " + cumuOptimal);
120     }
121 
122     /**
123      * Asks SidePanel to update PlayerMoves-gameStats.
124      *
125      * GUI uses this
126      */
127     public void setPlayerMoves() {
128         this.playerMoves.setText("Moves: " + p.getMovesPerformed());
129     }
130 
131     /**
132      * Asks SidePanel to update playerMovesSinceHit-gameStats.
133      *
134      * GUI uses this
135      */
136     public void setPlayerMovesSinceHit() {
137         this.playerMovesSinceHit.setText("Moves from last Goal: " + p.getMovesSinceHit());
138     }
139 
140     /**
141      * Asks SidePanel to update BehindOptimal-gameStats.
142      *
143      * GUI uses this
144      */
145     public void setBehindOptimal() {
146         int behindOptimal = (p.getMovesPerformed() - p.getMovesSinceHit()) - l.getOptimalMovesCumulative() + Math.max(0, p.getMovesSinceHit() - l.getOptimalMoves());
147         this.movesBehindTheOptimal.setText("You are " + behindOptimal + " behind the optimal");
148     }
149 
150     private void setPercentageOfExtraMoves() {
151         int behindOptimal = (p.getMovesPerformed() - p.getMovesSinceHit()) - l.getOptimalMovesCumulative() + Math.max(0, p.getMovesSinceHit() - l.getOptimalMoves());
152         int playerMovesLoc = p.getMovesPerformed();
153         if (playerMovesLoc == 0) {
154             this.percentageOfExtraMoves.setText("\n       0,00");
155         } else {
156             float percentage = (float) behindOptimal / playerMovesLoc * 100;
157             this.percentageOfExtraMoves.setText(String.format("\n       %.2f", percentage));
158         }
159     }
