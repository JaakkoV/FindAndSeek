10  public class DrawBoard extends JPanel {
11  
12      private Board gameboard;
13  
14      public DrawBoard(Board gameboard) {
15          this.gameboard = gameboard;
16      }
17  
18      @Override
19      public void paintComponent(Graphics g) {
20          Graphics2D g2d = (Graphics2D) g;
21          int width = 400;
22          int height = 400;
23  
24          int cellWidth = width / (this.gameboard.getWidth() + 1);
25          int cellHeight = height / (this.gameboard.getHeight() + 1);
26  
27          int xOffset = (width - (this.gameboard.getWidth() * cellWidth)) / 2;
28          int yOffset = (height - (this.gameboard.getHeight() * cellHeight)) / 2;
29          for (int i = 0; i < this.gameboard.getHeight(); i++) {
30              for (int j = 0; j < this.gameboard.getWidth(); j++) {
31                  if (isPlayer(i, j)) {
32                      Rectangle player = new Rectangle(xOffset + (j * cellWidth), yOffset + (i * cellHeight), cellWidth, cellHeight);
33                      g2d.setColor(Color.red);
34                      g2d.fill(player);
35                      g2d.setColor(Color.black);
36                      g2d.draw(player);
37                      char[] cs = {this.gameboard.getPlayer().getValue()};
38                      g2d.drawChars(cs, 0, 1, xOffset * 2 - 5 + (j * cellWidth), yOffset * 2 + (i * cellHeight)
39                      );
40                  } else if (isAnswer(i, j) != null) {
41                      Rectangle answer = new Rectangle(xOffset + (j * cellWidth), yOffset + (i * cellHeight), cellWidth, cellHeight);
42                      g2d.setColor(Color.yellow);
43                      g2d.fill(answer);
44                      g2d.setColor(Color.black);
45                      g2d.draw(answer);
46                      char[] cs = isAnswer(i, j);
47                      g2d.drawChars(cs, 0, 1, xOffset * 2 - 5 + (j * cellWidth), yOffset * 2 + (i * cellHeight));
48                  } else {
49                      Rectangle empty = new Rectangle(xOffset + (j * cellWidth), yOffset + (i * cellHeight), cellWidth, cellHeight);
50                      g2d.setColor(Color.gray);
51                      g2d.fill(empty);
52                      g2d.setColor(Color.black);
53                      g2d.draw(empty);
54                  }
55              }
56          }
57      }
58  
59      private boolean isPlayer(int i, int j) {
60          return this.gameboard.getPlayer().getLocation().equals(new Location(i, j));
61      }
62  
63      private char[] isAnswer(int i, int j) {
64          char [] returnsChar = new char[1];
65          for (Answer a : this.gameboard.getAnswers()) {
66              if (a.getLocation().equals(new Location(i, j))) {
67                  char ans = a.getValue();
68                  returnsChar[0] = ans;
69                  return returnsChar;
70              }
71          }
72          return null;
73      }
74  
75  }
